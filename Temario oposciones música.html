    // Cargar pregunta

    function loadQuestion() {

      const q = quizQuestions[currentQuestion];

      const quizBody = document.getElementById("quiz-body");

      let optionsHTML = "";

 

      if (q.type === "multiple") {

        optionsHTML = q.options.map((opt, index) =>

          `<button class="option" onclick="selectOption(this, ${index})">${opt}</button>`

        ).join('');

      } else if (q.type === "truefalse") {

        optionsHTML = `

          <button class="option" onclick="selectOption(this, 0)">Verdadero</button>

          <button class="option" onclick="selectOption(this, 1)">Falso</button>

        `;

      }

 

      quizBody.innerHTML = `

        <div class="question-box">

          <p><strong>Pregunta ${currentQuestion + 1} de 10:</strong></p>

          <p>${q.question}</p>

          <div class="options">

            ${optionsHTML}

          </div>

          <div id="feedback"></div>

        </div>

      `;

      document.getElementById("results").style.display = "none";

      document.getElementById("next-btn").style.display = "none";

    }

 

    // Seleccionar opción

    function selectOption(button, index) {

      const options = document.querySelectorAll(".option");

      options.forEach(opt => opt.classList.remove("selected"));

      button.classList.add("selected");

 

      const feedback = document.getElementById("feedback");

      const q = quizQuestions[currentQuestion];

      let isCorrect = false;

 

      if (q.type === "multiple") {

        isCorrect = index === q.correct;

      } else if (q.type === "truefalse") {

        isCorrect = (index === 0 && q.correct) || (index === 1 && !q.correct);

      }

 

      if (isCorrect) {

        feedback.innerHTML = `<div class="feedback correct">✅ ¡Correcto!</div>`;

        score++;

      } else {

        feedback.innerHTML = `<div class="feedback incorrect">❌ Incorrecto. ${q.explanation}</div>`;

      }

      document.getElementById("next-btn").style.display = "block";

    }

 

    // Siguiente pregunta

    function nextQuestion() {

      currentQuestion++;

      if (currentQuestion < quizQuestions.length) {

        loadQuestion();

      } else {

        showResults();

      }

    }

 

    // Mostrar resultados

    function showResults() {

      const resultsDiv = document.getElementById("results");

      const percentage = Math.round((score / 10) * 100);

      let message;

      if (percentage >= 80) {

        message = "🎉 ¡Excelente! Vas por muy buen camino.";

      } else if (percentage >= 60) {

        message = "👍 Buen trabajo. Repasa un poco más.";

      } else {

        message = "📚 Sigue estudiando. Cada intento te acerca al objetivo.";

      }

      resultsDiv.innerHTML = `

        <p>Tu puntuación: <strong>${score}/10</strong> (${percentage}%)</p>

        <p>${message}</p>

        <p><em>¡Cada vez que repitas el cuestionario, será diferente!</em></p>

      `;

      resultsDiv.style.display = "block";

      document.getElementById("quiz-body").innerHTML = "";

      document.getElementById("next-btn").style.display = "none";

    }
